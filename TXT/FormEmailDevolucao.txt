<script>
var _dadosPreview      = null;
var _jaEnviadas        = [];
var _comBase64         = '';   // base64 do arquivo do comunicado
var _comMime           = '';
var _comNome           = '';
// ── Comunicado — toggle ──────────────────────────────────────
function toggleComunicado() {
  var chk = document.getElementById('chk-comunicado').checked;
  document.getElementById('comunicado-fields').style.display = chk ? 'block' : 'none';
  if (!chk) removerComunicado();
}

// ── Comunicado — drop ────────────────────────────────────────
function onDropComunicado(evt) {
  evt.preventDefault();
  document.getElementById('drop-com').classList.remove('over');
  var f = evt.dataTransfer.files[0];
  if (f) processarArquivoComunicado(f);
}

function onFileComunicado(input) {
  if (input.files && input.files[0]) processarArquivoComunicado(input.files[0]);
}

function processarArquivoComunicado(file) {
  var maxBytes = 8 * 1024 * 1024;
  if (file.size > maxBytes) {
    mostrar('erro', '❌ Arquivo muito grande. O limite é 8 MB.');
    return;
  }
  var tipos = ['application/pdf','image/jpeg','image/jpg','image/png'];
  if (tipos.indexOf(file.type) === -1) {
    mostrar('erro', '❌ Formato inválido. Use PDF, JPG ou PNG.');
    return;
  }

  var reader = new FileReader();
  reader.onload = function(e) {
    var dataUrl = e.target.result;
    _comBase64  = dataUrl.split(',')[1];
    _comMime    = file.type;
    _comNome    = file.name;

    var drop = document.getElementById('drop-com');
    drop.classList.add('ok');
    drop.innerHTML = '✅ Arquivo selecionado — clique para trocar';

    var prev = document.getElementById('prev-com');
    document.getElementById('prev-com-nome').textContent = file.name;
    document.getElementById('prev-com-tam').textContent  =
      (file.size / 1024 < 1024)
        ? Math.round(file.size / 1024) + ' KB'
        : (file.size / 1048576).toFixed(1) + ' MB';
    prev.style.display = 'flex';
    document.getElementById('msg').style.display = 'none';
  };
  reader.readAsDataURL(file);
}

function removerComunicado() {
  _comBase64 = ''; _comMime = ''; _comNome = '';
  document.getElementById('file-comunicado').value = '';
  var drop = document.getElementById('drop-com');
  drop.classList.remove('ok');
  drop.innerHTML = '📎 Clique ou arraste o arquivo do comunicado aqui<br>' +
    '<span style="font-size:10px;opacity:.7">(PDF, JPG ou PNG — máx. 8 MB)</span>';
  document.getElementById('prev-com').style.display = 'none';
}

// ── Preview / Envio ──────────────────────────────────────────
function buscarPreview() {
  var nfds = document.getElementById('nfds').value.trim();
  if (!nfds) { mostrar('erro', '❌ Informe ao menos uma NFD.'); return; }

  document.getElementById('preview').style.display           = 'none';
  document.getElementById('btnEnviar').style.display         = 'none';
  document.getElementById('aviso-duplicado').style.display   = 'none';
  document.getElementById('btn-confirmar-envio').style.display = 'none';
  document.getElementById('msg').style.display               = 'none';
  document.getElementById('spinner').style.display           = 'block';
  document.getElementById('btnPreview').disabled             = true;

  google.script.run
    .withSuccessHandler(function(respCheck) {
      try { _jaEnviadas = JSON.parse(respCheck).jaEnviadas || []; } catch(_) { _jaEnviadas = []; }
      buscarPreviewDireto();
    })
    .withFailureHandler(function() { _jaEnviadas = []; buscarPreviewDireto(); })
    .verificarEmailsJaEnviados(nfds);
}

function buscarPreviewDireto() {
  var nfds = document.getElementById('nfds').value.trim();
  google.script.run
    .withSuccessHandler(function(resp) {
      document.getElementById('spinner').style.display  = 'none';
      document.getElementById('btnPreview').disabled    = false;
      var r = JSON.parse(resp);
      if (r.erro) { mostrar('erro', '❌ ' + r.erro); return; }
      _dadosPreview = r;

      if (r.emailsBase) {
        document.getElementById('emailsBase').innerHTML =
          '<strong>E-mails padrão (sempre incluídos):</strong> ' + r.emailsBase.join(', ');
      }
      if (!document.getElementById('assunto').value) {
        document.getElementById('assunto').value = r.titulo;
      }

      var tbody = document.getElementById('prevTabela');
      tbody.innerHTML = '';
      var total = 0;
      r.itens.forEach(function(it) {
        var badge = it.tipo === 'Avaria'   ? '<span class="badge-a">Avaria</span>'
                  : it.tipo === 'Falta'    ? '<span class="badge-f">Falta</span>'
                  : it.tipo === 'Rejeição' ? '<span class="badge-r">Rejeição</span>'
                  : it.tipo || '—';
        var val = it.valor > 0 ? 'R$ ' + it.valor.toFixed(2).replace('.',',') : '—';
        total += it.valor;
        var anexoIcon = it.urlAnexo
          ? ' <span title="NF anexada" style="color:#2E7D32">📎</span>' : '';
        tbody.innerHTML +=
          '<tr><td><b>' + esc(it.nfd) + '</b>' + anexoIcon + '</td>' +
          '<td>' + badge + '</td>' +
          '<td style="color:#555;font-size:11px">' + esc((it.motivo||'—').substring(0,35)) + '</td>' +
          '<td>' + esc(it.nf) + '</td>' +
          '<td>' + esc(it.desc.substring(0,35)) + '</td>' +
          '<td style="text-align:center">' + it.qtd + '</td>' +
          '<td style="text-align:right">' + val + '</td>' +
          '<td style="text-align:center">' + esc(it.data) + '</td></tr>';
      });

      document.getElementById('prevTotal').innerHTML =
        '<td colspan="6" style="text-align:right;padding:6px 8px">TOTAL:</td>' +
        '<td style="text-align:right;padding:6px 8px;color:#25419A">R$ ' +
        total.toFixed(2).replace('.',',') + '</td><td></td>';

      document.getElementById('prevAssunto').innerHTML =
        '<strong>Assunto:</strong> ' + esc(document.getElementById('assunto').value || r.titulo) +
        ' &nbsp;|&nbsp; <strong>Fornecedor:</strong> ' + esc(r.forn) +
        ' &nbsp;|&nbsp; <strong>' + r.itens.length + ' NFD(s)</strong>';

      document.getElementById('preview').style.display = 'block';

      if (_jaEnviadas.length > 0) {
        var avisoEl = document.getElementById('aviso-duplicado');
        var linhas  = _jaEnviadas.map(function(j) {
          return '• <strong>' + esc(j.nfd) + '</strong>: já enviada ' + j.total +
                 ' vez(es) — último envio em <strong>' + esc(j.data) + '</strong>';
        });
        avisoEl.innerHTML =
          '⚠️ <strong>Atenção: e-mail já enviado para as seguintes NFDs:</strong><br>' +
          linhas.join('<br>') + '<br><br>Deseja enviar novamente mesmo assim?';
        avisoEl.style.display                              = 'block';
        document.getElementById('btnEnviar').style.display          = 'none';
        document.getElementById('btn-confirmar-envio').style.display = 'block';
      } else {
        document.getElementById('btnEnviar').style.display           = 'block';
        document.getElementById('btn-confirmar-envio').style.display = 'none';
        document.getElementById('aviso-duplicado').style.display     = 'none';
      }
    })
    .withFailureHandler(function(e) {
      document.getElementById('spinner').style.display  = 'none';
      document.getElementById('btnPreview').disabled    = false;
      mostrar('erro', '❌ ' + e.message);
    })
    .buscarDadosNFDs(nfds);
}

function enviar(forcado) {
  document.getElementById('btnEnviar').disabled             = true;
  document.getElementById('btn-confirmar-envio').disabled   = true;
  document.getElementById('spinner').style.display          = 'block';
  document.getElementById('msg').style.display              = 'none';
  document.getElementById('aviso-duplicado').style.display  = 'none';

  var usaComunicado = document.getElementById('chk-comunicado').checked;
  if (usaComunicado && !_comBase64) {
    mostrar('erro', '❌ Selecione o arquivo do comunicado de retorno antes de enviar.');
    document.getElementById('btnEnviar').disabled           = false;
    document.getElementById('btn-confirmar-envio').disabled = false;
    document.getElementById('spinner').style.display        = 'none';
    return;
  }

  var params = {
    nfdsRaw:          document.getElementById('nfds').value.trim(),
    emailsExtras:     document.getElementById('emailsExtras').value.trim(),
    assunto:          document.getElementById('assunto').value.trim(),
    obs:              document.getElementById('obs').value.trim(),
    comBase64:        usaComunicado ? _comBase64 : '',
    comMime:          usaComunicado ? _comMime   : '',
    comNome:          usaComunicado ? _comNome   : '',
    comObs:           usaComunicado ? document.getElementById('obs-comunicado').value.trim() : ''
  };

  google.script.run
    .withSuccessHandler(function(resp) {
      document.getElementById('spinner').style.display          = 'none';
      document.getElementById('btnEnviar').disabled             = false;
      document.getElementById('btn-confirmar-envio').disabled   = false;
      var r = JSON.parse(resp);
      if (r.sucesso) {
        mostrar('ok', r.sucesso);
        document.getElementById('preview').style.display             = 'none';
        document.getElementById('btnEnviar').style.display           = 'none';
        document.getElementById('btn-confirmar-envio').style.display = 'none';
        // Limpar todos os campos
        ['nfds','assunto','obs','emailsExtras','obs-comunicado'].forEach(function(id){
          document.getElementById(id).value = '';
        });
        document.getElementById('chk-comunicado').checked = false;
        document.getElementById('comunicado-fields').style.display = 'none';
        removerComunicado();
        _jaEnviadas = [];
      } else {
        mostrar('erro', r.erro || 'Erro desconhecido.');
        document.getElementById('btnEnviar').style.display = 'block';
      }
    })
    .withFailureHandler(function(e) {
      document.getElementById('spinner').style.display          = 'none';
      document.getElementById('btnEnviar').disabled             = false;
      document.getElementById('btn-confirmar-envio').disabled   = false;
      mostrar('erro', '❌ ' + e.message);
    })
    .enviarEmailDevolucao(params);
}

function mostrar(tipo, txt) {
  var el = document.getElementById('msg');
  el.className    = 'msg-' + tipo;
  el.textContent  = txt;
