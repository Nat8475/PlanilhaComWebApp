<script>
var _data = null;
var _forn = 'TODOS';

var _fmt = function(v) {
  return 'R$ ' + (parseFloat(v)||0).toLocaleString('pt-BR',{minimumFractionDigits:2,maximumFractionDigits:2});
};
var _esc = function(s) {
  return String(s||'').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');
};

function carregar() {
  document.getElementById('ts-update').textContent = 'Atualizando...';
  try {
    google.script.run
      .withSuccessHandler(function(resp) {
        try {
          var d = JSON.parse(resp);
          if (d.erro) { document.getElementById('ts-update').textContent = 'Erro: ' + d.erro; return; }
          _data = d;
          renderAll();
        } catch(e) { document.getElementById('ts-update').textContent = 'Erro ao processar.'; }
      })
      .withFailureHandler(function(e) { document.getElementById('ts-update').textContent = 'Erro: ' + e.message; })
      .obterDadosDashboard();
  } catch(e) { document.getElementById('ts-update').textContent = 'Erro.'; }
}

function renderAll() {
  if (!_data) return;
  renderTabs();
  renderKpis();
  renderChartBars();
  renderDonut();
  renderTabela();
  document.getElementById('ts-update').textContent =
    'Atualizado às ' + new Date().toLocaleTimeString('pt-BR',{hour:'2-digit',minute:'2-digit'});
}

function renderTabs() {
  var fornArr = _data.porFornecedor || [];
  var total = 0;
  fornArr.forEach(function(f) { total += f.total; });

  var html = '<div class="forn-tab on" data-f="TODOS" onclick="selForn(this)">'
    + 'Todos <span class="forn-badge">'+total+'</span></div>';
  fornArr.forEach(function(f) {
    html += '<div class="forn-tab" data-f="'+_esc(f.forn)+'" onclick="selForn(this)">'
      + _esc(f.forn) + ' <span class="forn-badge">'+f.total+'</span></div>';
  });
  document.getElementById('forn-tabs').innerHTML = html;
}

function selForn(el) {
  document.querySelectorAll('.forn-tab').forEach(function(t){ t.classList.remove('on'); });
  el.classList.add('on');
  _forn = el.dataset.f;
  renderKpis();
  renderTabela();
}

function _getFornData() {
  if (_forn === 'TODOS') {
    var c = _data.counts || {};
    return {Pendente:c.Pendente||0, EmTransferencia:c.EmTransferencia||0,
            Devolvido:c.Devolvido||0, Venda:c.Venda||0,
            atrasos:_data.atrasos30||0, vlTot:(_data.valores||{}).Pendente||0};
  }
  var pf = (_data.porFornecedor||[]).filter(function(f){ return f.forn===_forn; });
  return pf.length ? pf[0] : {Pendente:0,EmTransferencia:0,Devolvido:0,Venda:0,atrasos:0,vlTot:0};
}

function renderKpis() {
  var f = _getFornData();
  var defs = [
    {l:'Pendentes',        n:f.Pendente||0,        ac:'#2563EB', s:'aguardando resolução'},
    {l:'Em Transferência', n:f.EmTransferencia||0,  ac:'#0891B2', s:'em trânsito'},
    {l:'Devolvidos',       n:f.Devolvido||0,        ac:'#059669', s:'devoluções concluídas'},
    {l:'Baixa p/ Venda',  n:f.Venda||0,            ac:'#D97706', s:'enviados para venda'},
    {l:'Atrasos +30d',    n:f.atrasos||0,           ac:'#DC2626', s:'pendentes há mais de 30 dias'}
  ];
  document.getElementById('kpi-row').innerHTML = defs.map(function(k) {
    return '<div class="kpi" style="--kac:'+k.ac+'">'
      + '<div class="kl">'+k.l+'</div>'
      + '<div class="kn">'+k.n+'</div>'
      + '<div class="ks">'+k.s+'</div>'
      + '</div>';
  }).join('');
}

function renderChartBars() {
  var fornArr = (_data.porFornecedor||[]).slice(0, 8);
  if (!fornArr.length) { document.getElementById('chart-bars').innerHTML = '<div class="empty">Sem dados.</div>'; return; }
  var max = 1;
  fornArr.forEach(function(f){ if (f.total > max) max = f.total; });

  document.getElementById('chart-bars').innerHTML = fornArr.map(function(f) {
    var bars = [
      {n:f.Pendente||0, c:'#2563EB'},
      {n:f.EmTransferencia||0, c:'#0891B2'},
      {n:f.Devolvido||0, c:'#059669'},
      {n:f.Venda||0, c:'#D97706'}
    ];
    var seg = '';
    var tot = f.total || 1;
    bars.forEach(function(b){
      if (!b.n) return;
      seg += '<div class="bar-fill" style="background:'+b.c+';width:'+(b.n/tot*100).toFixed(1)+'%;float:left"></div>';
    });
    return '<div class="bar-row">'
      + '<div class="bar-label" title="'+_esc(f.forn)+'">'+_esc(f.forn.length>12?f.forn.slice(0,10)+'…':f.forn)+'</div>'
      + '<div class="bar-track" title="'+f.total+' itens">'+seg+'</div>'
      + '<div class="bar-num">'+f.total+'</div>'
      + '</div>';
  }).join('');
}

function renderDonut() {
  var c = _data.counts || {};
  var segs = [
    {l:'Pendente',        n:c.Pendente||0,        color:'#2563EB'},
    {l:'Em Transferência',n:c.EmTransferencia||0,  color:'#0891B2'},
    {l:'Devolvido',       n:c.Devolvido||0,        color:'#059669'},
    {l:'Venda',           n:c.Venda||0,            color:'#D97706'}
  ].filter(function(s){ return s.n > 0; });

  var tot = segs.reduce(function(a,s){ return a+s.n; }, 0) || 1;
  var r = 54, cx = 65, cy = 65, stroke = 22;
  var circum = 2 * Math.PI * r;
  var off = 0;
  var svgSegs = segs.map(function(s) {
    var pct = s.n / tot;
    var dash = circum * pct;
    var el = '<circle cx="'+cx+'" cy="'+cy+'" r="'+r+'" fill="none" stroke="'+s.color+'"'
      + ' stroke-width="'+stroke+'" stroke-dasharray="'+(dash-1)+' '+(circum-dash+1)+'"'
      + ' stroke-dashoffset="'+(-off)+'" transform="rotate(-90 '+cx+' '+cy+')">'
      + '<title>'+s.l+': '+s.n+'</title></circle>';
    off += dash;
    return el;
  }).join('');
  var svg = '<svg class="donut-svg" width="130" height="130" viewBox="0 0 130 130">'
    + '<circle cx="'+cx+'" cy="'+cy+'" r="'+r+'" fill="none" stroke="#F3F4F6" stroke-width="'+stroke+'"/>'
    + svgSegs
    + '<text x="'+cx+'" y="'+(cy+5)+'" text-anchor="middle" font:700 18px/1 var(--font)" fill="#111827" style="font:700 18px/1 sans-serif">'+tot+'</text>'
    + '</svg>';
  var legend = '<div class="donut-legend">' + segs.map(function(s){
    return '<div class="donut-leg-row">'
      + '<div class="donut-leg-dot" style="background:'+s.color+'"></div>'
      + '<span>'+s.l+'</span>'
      + '<span class="donut-leg-val">'+s.n+'</span></div>';
  }).join('') + '</div>';
  document.getElementById('donut-wrap').innerHTML = svg + legend;
}

function renderTabela() {
  var bm = {'Pendente':'b-pend','Em Transferência':'b-tr','Devolvido':'b-dev','Venda':'b-venda'};
  var recentes = (_data.recentes || []).filter(function(n){
    return _forn === 'TODOS' || n.forn === _forn;
  });

  var title = _forn === 'TODOS' ? 'Lançamentos Recentes' : 'Lançamentos — ' + _forn;
  document.getElementById('tbl-title').textContent = title;

  var rows = recentes.map(function(n) {
    var bc = bm[n.status] || 'b-pend';
    return '<tr>'
      + '<td><span class="nf-code">'+(n.nfd||n.nf)+'</span></td>'
      + '<td>'+_esc(n.forn)+'</td>'
      + '<td style="max-width:160px;overflow:hidden;text-overflow:ellipsis;white-space:nowrap;color:var(--text-sm)">'+_esc(n.desc||'—')+'</td>'
      + '<td style="text-align:center;color:var(--text-sm)">'+((n.qtd||0)||'—')+'</td>'
      + '<td style="text-align:right;font-variant-numeric:tabular-nums">'+_fmt(n.vlTot)+'</td>'
      + '<td><span class="badge '+bc+'">'+n.status+'</span></td>'
      + '<td style="color:var(--text-sm);white-space:nowrap">'+_esc(n.data)+'</td>'
      + '</tr>';
  }).join('');
  document.getElementById('tbl-body').innerHTML = rows ||
    '<tr><td colspan="7" class="empty">Nenhum lançamento para este fornecedor.</td></tr>';
}

carregar();
</script>
