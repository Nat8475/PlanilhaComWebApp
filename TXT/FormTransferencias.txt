<script>
var _itens = [], _filt = [], _sel = [];
var _idxAcao = -1, _linhaBaixa = 0, _linhaCancel = 0, _linhaReag = 0;

function carregar() {
  document.getElementById('spn').style.display = 'block';
  document.getElementById('tbl').style.display = 'none';
  document.getElementById('empty').style.display = 'none';
  _itens = []; _filt = []; _sel = [];
  atualizarBulk();

  google.script.run
    .withSuccessHandler(function(resp) {
      document.getElementById('spn').style.display = 'none';
      try {
        var r = JSON.parse(resp);
        if (r.erro) { toast(r.erro, 'err'); return; }
        _itens = r.itens || [];
        document.getElementById('uatt').textContent = 'Atualizado: ' + new Date().toLocaleTimeString('pt-BR');
        filtrar();
      } catch(e) { toast('Erro: ' + e.message, 'err'); }
    })
    .withFailureHandler(function(e) {
      document.getElementById('spn').style.display = 'none';
      toast('Erro: ' + e.message, 'err');
    })
    .obterTransferencias({});
}

function filtrar() {
  var fs = document.getElementById('fst').value;
  var fb = (document.getElementById('fbusca').value || '').toLowerCase().trim();
  _filt = _itens.filter(function(it) {
    if (fs && it.stTransf !== fs) return false;
    if (fb) {
      var h = ((it.nf||'')+(it.nfd||'')+(it.forn||'')+(it.transportadora||'')+(it.abaOrigem||'')).toLowerCase();
      if (h.indexOf(fb) === -1) return false;
    }
    return true;
  });
  _sel = new Array(_filt.length).fill(false);
  renderTabela();
  atualizarBulk();
  var em = _itens.filter(function(it){ return it.stTransf === 'Em Transferência'; }).length;
  var venc = _itens.filter(function(it){ return it.atrasado; }).length;
  document.getElementById('bcnt').textContent =
    em + ' em transf.' + (venc > 0 ? ' · ' + venc + ' vencida(s)' : '');
}

function renderTabela() {
  var tb = document.getElementById('tbody');
  if (!_filt.length) {
    document.getElementById('tbl').style.display = 'none';
    document.getElementById('empty').style.display = 'block';
    return;
  }
  document.getElementById('tbl').style.display = 'table';
  document.getElementById('empty').style.display = 'none';
  var hoje = new Date(); hoje.setHours(0,0,0,0);

  tb.innerHTML = _filt.map(function(it, i) {
    var ativo = it.stTransf === 'Em Transferência';
    var agendDt = it.dataAgend ? parseDtBR(it.dataAgend) : null;
    var diasTag = '';
    if (ativo && agendDt) {
      var diff = Math.ceil((agendDt - hoje) / 864e5);
      if (diff < 0)    diasTag = '<span class="dias-late">' + Math.abs(diff) + 'd atr.</span>';
      else if (diff<=2) diasTag = '<span class="dias-warn">em ' + diff + 'd</span>';
      else              diasTag = '<span class="dias-ok">em ' + diff + 'd</span>';
    }
    var cls = it.atrasado ? 'trat' : (it.stTransf==='Concluída' ? 'tconc' : it.stTransf==='Cancelada' ? 'tcanc' : '');
    var badge = it.stTransf === 'Em Transferência'
      ? '<span class="bdg btr">Em Transferência</span>'
      : it.stTransf === 'Concluída'
      ? '<span class="bdg bconc">Concluída</span>'
      : '<span class="bdg bcanc">Cancelada</span>';
    var fvStr = it.freteValor > 0 ? 'R$ ' + it.freteValor.toLocaleString('pt-BR', {minimumFractionDigits:2}) : '—';
    var acoes = ativo
      ? '<div class="acoes-td">'
        + '<button class="bact bgreen" onclick="abrirBaixa(' + i + ')">&#9989; Baixa</button>'
        + '<button class="bact bcyan" style="background:var(--cyan-bg);color:var(--cyan)" onclick="abrirReagendar(' + i + ')">&#128197;</button>'
        + '<button class="bact bred" style="background:var(--red-bg);color:var(--red)" onclick="abrirCancelar(' + i + ')">&#10006;</button>'
        + '</div>'
      : '<span style="font-size:10px;color:var(--text-sm)">' + esc(it.dataBaixa||it.obsCancel||'') + '</span>';

    return '<tr class="' + cls + '">'
      + '<td><input type="checkbox"' + (!ativo ? ' disabled' : '') + (ativo ? ' onchange="toggleItem(' + i + ',this)"' : '') + '></td>'
      + '<td><b style="color:var(--navy);font-size:11px">' + esc(it.nf) + '</b><div style="font-size:10px;color:var(--text-sm)">' + esc(it.nfd) + '</div></td>'
      + '<td>' + esc(it.forn) + '</td>'
      + '<td style="color:var(--cyan);font-weight:600">' + esc(it.transportadora || '—') + '</td>'
      + '<td style="font-size:11px;color:var(--text-sm)">' + esc(it.abaOrigem) + '</td>'
      + '<td style="font-size:11px">' + esc(it.freteTipo) + '<br><span style="color:var(--text-sm)">' + fvStr + '</span></td>'
      + '<td>' + esc(it.dataAgend||'—') + ' ' + diasTag + '</td>'
      + '<td style="text-align:center"><span style="font-size:10px;color:var(--text-sm)">' + (it.diasEmTransf || 0) + 'd</span></td>'
      + '<td>' + badge + '</td>'
      + '<td>' + acoes + '</td>'
      + '</tr>';
  }).join('');
}

function toggleItem(i, chk) { _sel[i] = chk.checked; atualizarBulk(); }
function toggleTodos(s) {
  _sel = _filt.map(function(it){ return s && it.stTransf === 'Em Transferência'; });
  document.querySelectorAll('#tbody input[type=checkbox]').forEach(function(c,i){ if (!c.disabled) c.checked = _sel[i]; });
  atualizarBulk();
}
function limparSel() {
  _sel = new Array(_filt.length).fill(false);
  document.querySelectorAll('#tbody input[type=checkbox]').forEach(function(c){ c.checked = false; });
  document.getElementById('chkAll').checked = false;
  atualizarBulk();
}
function atualizarBulk() {
  var cnt = _sel.filter(Boolean).length;
  document.getElementById('blkbar').style.display = cnt > 0 ? 'flex' : 'none';
  document.getElementById('blkcnt').textContent = cnt + ' selecionada(s)';
}

// ── Baixa ─────────────────────────────────────────────────────
function abrirBaixa(i) {
  _idxAcao = i;
  _linhaBaixa = _filt[i].linha;
  document.getElementById('mBaixaTitulo').textContent = '✅ Confirmar Baixa — NF ' + _filt[i].nf;
  document.getElementById('mBaixaDesc').textContent = 'Fornecedor: ' + _filt[i].forn
    + ((_filt[i].transportadora) ? ' · Transp.: ' + _filt[i].transportadora : '');
  document.getElementById('bObs').value = '';
  document.getElementById('bArq').value = '';
  document.getElementById('mBaixaMsg').className = 'mmsg';
  document.getElementById('mBaixa').style.display = 'flex';
}

function confirmarBaixa() {
  var btn = document.getElementById('btnConfBaixa');
  btn.disabled = true; btn.textContent = '⏳';
  var obs   = document.getElementById('bObs').value.trim();
  var arqEl = document.getElementById('bArq');
  var arq   = arqEl.files && arqEl.files[0];
  var params = { linha: _linhaBaixa, obs: obs };

  function enviar() {
    google.script.run
      .withSuccessHandler(function(resp) {
        btn.disabled = false; btn.textContent = '✅ Confirmar Baixa';
        try {
          var r = JSON.parse(resp);
          if (r.ok) { fecharModais(); toast(r.ok, 'ok'); setTimeout(carregar, 800); }
          else { showMsg('mBaixaMsg', r.erro || 'Erro.', 'err'); }
        } catch(e) { showMsg('mBaixaMsg', e.message, 'err'); }
      })
      .withFailureHandler(function(e) { btn.disabled = false; btn.textContent = '✅ Confirmar Baixa'; showMsg('mBaixaMsg', e.message, 'err'); })
      .darBaixaTransferencia(params);
  }

  if (arq) {
    var reader = new FileReader();
    reader.onload = function(ev) {
      var b64 = ev.target.result.split(',')[1];
      params.base64      = b64;
      params.mimeType    = arq.type || 'application/octet-stream';
      params.nomeArquivo = arq.name;
      enviar();
    };
    reader.onerror = function(){ showMsg('mBaixaMsg', 'Erro ao ler arquivo.', 'err'); btn.disabled = false; btn.textContent = '✅ Confirmar Baixa'; };
    reader.readAsDataURL(arq);
  } else {
    enviar();
  }
}

// ── Cancelar ──────────────────────────────────────────────────
function abrirCancelar(i) {
  _idxAcao = i;
  _linhaCancel = _filt[i].linha;
  document.getElementById('mCancelTitulo').textContent = '❌ Cancelar — NF ' + _filt[i].nf;
  document.getElementById('mCancelDesc').textContent = 'Fornecedor: ' + _filt[i].forn;
  document.getElementById('cObs').value = '';
  document.getElementById('mCancelMsg').className = 'mmsg';
  document.getElementById('mCancel').style.display = 'flex';
}

function abrirCancelarLote() {
  _linhaCancel = null; // sinaliza lote
  var sel = _filt.filter(function(_,i){ return _sel[i]; });
  document.getElementById('mCancelTitulo').textContent = '❌ Cancelar ' + sel.length + ' Transferência(s)';
  document.getElementById('mCancelDesc').textContent = sel.map(function(it){ return 'NF ' + it.nf; }).join(', ');
  document.getElementById('cObs').value = '';
  document.getElementById('mCancelMsg').className = 'mmsg';
  document.getElementById('mCancel').style.display = 'flex';
}

function confirmarCancelar() {
  var obs = document.getElementById('cObs').value.trim();
  if (!obs) { showMsg('mCancelMsg', 'O motivo do cancelamento é obrigatório.', 'err'); return; }
  var btn = document.getElementById('btnConfCancel');
  btn.disabled = true; btn.textContent = '⏳';

  if (_linhaCancel) {
    // Cancelamento único
    google.script.run
      .withSuccessHandler(function(resp) {
        btn.disabled = false; btn.textContent = '❌ Confirmar Cancelamento';
        try {
          var r = JSON.parse(resp);
          if (r.ok) { fecharModais(); toast(r.ok, 'ok'); setTimeout(carregar, 800); }
          else { showMsg('mCancelMsg', r.erro || 'Erro.', 'err'); }
        } catch(e) { showMsg('mCancelMsg', e.message, 'err'); }
      })
      .withFailureHandler(function(e) { btn.disabled = false; btn.textContent = '❌ Confirmar Cancelamento'; showMsg('mCancelMsg', e.message, 'err'); })
      .cancelarTransferencia({ linha: _linhaCancel, obs: obs });
  } else {
    // Cancelamento em lote (chama uma a uma em sequência)
    var linhas = _filt.filter(function(_,i){ return _sel[i]; }).map(function(it){ return it.linha; });
    var ok = 0, errs = [];
    var idx = 0;
    function next() {
      if (idx >= linhas.length) {
        btn.disabled = false; btn.textContent = '❌ Confirmar Cancelamento';
        fecharModais(); limparSel();
        toast('✅ ' + ok + ' cancelada(s)' + (errs.length ? ' · ' + errs.length + ' erro(s)' : ''), ok > 0 ? 'ok' : 'err');
        setTimeout(carregar, 800);
        return;
      }
      google.script.run
        .withSuccessHandler(function(resp) {
          try { var r = JSON.parse(resp); if (r.ok) ok++; else errs.push(r.erro); } catch(e) { errs.push(e.message); }
          idx++; next();
        })
        .withFailureHandler(function(e) { errs.push(e.message); idx++; next(); })
        .cancelarTransferencia({ linha: linhas[idx], obs: obs });
    }
    next();
  }
}

// ── Reagendar ─────────────────────────────────────────────────
function abrirReagendar(i) {
  _idxAcao = i;
  _linhaReag = _filt[i].linha;
  document.getElementById('mReagDesc').textContent = 'NF ' + _filt[i].nf + ' · ' + _filt[i].forn;
  var dAtual = _filt[i].dataAgend ? parseDtBR(_filt[i].dataAgend) : null;
  document.getElementById('rData').value = dAtual ? dAtual.toISOString().split('T')[0] : '';
  document.getElementById('rObs').value = '';
  document.getElementById('mReagMsg').className = 'mmsg';
  document.getElementById('mReag').style.display = 'flex';
}

function confirmarReagendar() {
  var nova = document.getElementById('rData').value;
  if (!nova) { showMsg('mReagMsg', 'Selecione a nova data.', 'err'); return; }
  var btn = document.getElementById('btnConfReag');
  btn.disabled = true; btn.textContent = '⏳';
  google.script.run
    .withSuccessHandler(function(resp) {
      btn.disabled = false; btn.textContent = '📅 Confirmar';
      try {
        var r = JSON.parse(resp);
        if (r.ok) { fecharModais(); toast(r.ok, 'ok'); setTimeout(carregar, 800); }
        else { showMsg('mReagMsg', r.erro || 'Erro.', 'err'); }
      } catch(e) { showMsg('mReagMsg', e.message, 'err'); }
    })
    .withFailureHandler(function(e) { btn.disabled = false; btn.textContent = '📅 Confirmar'; showMsg('mReagMsg', e.message, 'err'); })
    .reagendarTransferencia({ linha: _linhaReag, dataAgendamento: nova, obs: document.getElementById('rObs').value.trim() });
}

// ── Helpers ───────────────────────────────────────────────────
function fecharModais() {
  ['mBaixa','mCancel','mReag'].forEach(function(id){ document.getElementById(id).style.display='none'; });
  _linhaCancel = 0; _linhaBaixa = 0; _linhaReag = 0;
}
function showMsg(id, txt, tipo) {
  var el = document.getElementById(id);
  el.textContent = txt; el.className = 'mmsg ' + tipo;
}
function parseDtBR(s) {
  if (!s) return null;
  var p = s.split('/'); if (p.length !== 3) return null;
  return new Date(+p[2], +p[1]-1, +p[0]);
}
function esc(s) {
  return String(s==null?'':s)
    .replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');
}
function toast(msg, tipo) {
  var el = document.getElementById('toast');
  el.className = 'toast tst-' + (tipo||'ok');
  el.textContent = msg; el.style.display = 'block';
  clearTimeout(el._t);
  el._t = setTimeout(function(){ el.style.display = 'none'; }, 4500);
}
document.addEventListener('keydown', function(e) {
  if (e.key === 'Escape') fecharModais();
});
